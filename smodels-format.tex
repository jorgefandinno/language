\begin{frame}{\smodels\ format}
  \medskip
  \begin{itemize}
  \item The \smodels\ format consists of
    \begin{itemize}
    \item normal rules
    \item choice rules
    \item cardinality rules
    \item weight rules
    \item minimization statements
    \end{itemize}
    \medskip
  \item Block-oriented format
  \end{itemize}
\end{frame}
